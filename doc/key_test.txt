a) '#0' -- Information, if the following keys are pressed
               '*0' -- Display MAC address

b) '#1' -- Display test, if the following keys are pressed
               '*0' -- Display ASCII font
               '*1' -- Display Chinese font
               '*2' -- Next page
               '*3' -- Previous page
               '*4' -- BMP 1 display/Lcd Test 4
               '*5' -- BMP 1 reverse display/Lcd Test 5
               '*6' -- BMP 2 display
               '*7' -- BMP 2 reverse display
               '*8' -- BMP 3 display
               '*9' -- BMP 3 reverse display

c) '#2' -- Speed test, if the following keys are pressed
               '*0' -- Test flash read speed
			   '*1' -- Test lcd display speed
			   '*2" -- Test delay function speed
			
d) '#3' -- DSP test, if the following keys are pressed
               '*0' -- Test g.729 DSP_OP_TEST_ENCODE & DSP_OP_TEST_DECODE
			   '*9' -- Test DM last 256 words

e) '#4' -- Ethernet test, if the following keys are pressed
               '*0' -- Test NE2000 buffer read and write

f) '#5' -- System test, if the following keys are pressed
               '*0' -- Restore MAC and Product ID, and set common settings
               '*1' -- Copy current settings to default settings
               '*5' -- Copy default settings to current settings
	       '*9' -- Clear config_id to force auto-provision

g) '#6' -- CODEC test, if the following keys are pressed
               '*0' -- Start analog loop back
               '*1' -- Stop analog loop back
               '*2' -- Start 8k digital loop back
               '*3' -- Stop 8k digital loop back    
               '*4' -- Start 8k digital loop back, handfree mode, *3 to stop too

j) '#9' -- Run mode control, if the following keys are pressed
               '*0' -- Restart in page0
